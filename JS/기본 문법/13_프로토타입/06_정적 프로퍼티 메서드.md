```js
function Person(name){
   this.name = name
}

Person.staticMethod = function(){
   console.log('yes');
}
const me = new Person('sa')
me.staticMethod() // error 정적메서드이기 때문에 me의 프로퍼티체인에 존재하지 않는다.
Person.staticMethod() // yes
```
